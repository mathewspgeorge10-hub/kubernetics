cd k8s
kubectl create deployment node-docker --image=urk24cs7151/node-docker:1.0 --dry-run=client -o yaml > deployment.yaml
kubectl apply -f deployment.yaml
kubectl expose deployment node-docker --port=80 --target-port=3000 --type=LoadBalancer --dry-run=client -o yaml > service.yaml
kubectl apply -f service.yaml
kubectl get deployments
kubectl get services
kubectl get pods
kubectl get all
minikube service node-docker --url
